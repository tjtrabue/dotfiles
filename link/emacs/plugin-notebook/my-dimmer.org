=dimmer= dynamically adjusts inactive windows in Emacs to be dimmer than the
active window.

#+begin_src emacs-lisp :tangle yes
  (use-package dimmer
      :config
      (dimmer-configure-which-key)
      (dimmer-configure-helm)
      ;; dimmer messes with lsp-mode
      ;; (dimmer-mode t)
  )
#+end_src
