* ansible
#+begin_src emacs-lisp

#+end_src

* docker
#+begin_src emacs-lisp
(use-package docker
  :bind
  (("C-c d" . docker)))
#+end_src

* kubernetes
  A magit-style interface to the Kubernetes command-line client.
  There is also a =kubernetes-evil= plugin for evil-mode support.
  See =my-evil= for its configuration.

#+begin_src emacs-lisp
  (use-package kubernetes
    :commands
    (kubernetes-overview))
#+end_src
